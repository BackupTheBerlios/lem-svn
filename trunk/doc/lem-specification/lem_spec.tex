% Embed a code fragment in the document
% Parameters:
% * The file to include
% * The caption
% * The scale
% Example: \codefrag{object}{Object creation}{0.5}
\newcommand{\codefrag}[3]{
	\begin{figure}[!ht]
		\begin{center}
			\framebox{\includegraphics[scale=#3]{snippets/#1.eps}}
			\caption{#2}
			\label{frag:#1}
		\end{center}
	\end{figure}
}

\newcommand{\sidebysidefrag}[4]{
	\begin{figure}[!ht]
		\begin{center}
			\framebox{\includegraphics[scale=#4]{snippets/#1.eps}}
			\framebox{\includegraphics[scale=#4]{snippets/#2.eps}}
			\caption{#3}
			\label{frag:#1}
		\end{center}
	\end{figure}
}

% Embed a dia diagram in the document
% Parameters:
% * The file to include
% * The caption
% * The scale
% * The rotation
\newcommand{\diadiagram}[4]{
	\begin{figure}[!ht]
		\begin{center}
			\includegraphics[scale=#3,angle=#4]{images/#1.eps}
			\caption{#2}
			\label{frag:#1}
		\end{center}
	\end{figure}
}
